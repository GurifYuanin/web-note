实现 Promise

实现基本功能

  1. new Promise 传入执行函数并立即执行
  2. 执行函数控制更新 Promise 的状态
  3. then / catch 添加回调函数进入队列，若已经是最终态（resolved / rejected）则直接执行
  4. 当 Promise 状态更新时，执行所有未执行的回调函数队列
  5. then 支持链式调用，且每次调用的返回值都是一个新的 Promise

class MyPromise {
  static PENDING = 'pending';
  static RESOLVED = 'resolved';
  static REJECTED = 'rejected';

  status = MyPromise.PENDING;
  value = null;
  reason = null;
  resolvedQueues = [];
  rejectedQueues = [];

  constructor(executor) {
    const resolve = (value) => {
      if (this.status === MyPromise.PENDING) {
        this.status = MyPromise.RESOLVED;
        this.value = value;
        this.resolvedQueues.forEach(fn => fn(value));
      }
    }
    const reject = (reasone) => {
      if (this.status === MyPromise.PENDING) {
        this.status = MyPromise.rejected;
        this.reason = reason;
        this.rejectedQueues.forEach(fn => fn(reason));
      }
    }
    executor(resolve, reject);
  }

  then(onFulfilled, onRejected) {
    onFulfilled = typeof onFulfilled === 'function' ? onFulfilled : v => v;
    onRejected = typeof onRejected === 'function' ? onRejected : (reason) => { throw reason };
    return new MyPromise((resolve, rejected) => {
      if (this.status === MyPromise.PENDING) {
        this.resolvedQueues.push(() => {
          resolve(onFulfilled(this.value));
        });
        this.rejectedQueues.push(() => {
          reject(onRejected(this.reason));
        });
      }
      if (this.status === MyPromise.RESOLVED) {
        resolve(onFulfilled(this.value));
      }
      if (this.status === MyPromise.REJECTED) {
        reject(onRejected(this.reason));
      }
    });
  }
}
const promise = new MyPromise((resolve) => {
  console.log('new MyPromise');
  setTimeout(() => {
    console.log('resolve');
    resolve('resolve');
  }, 1000);
});
promise.then(data => {
  console.log('then1: ' + data);
  return 'then1';
}).then(data => {
  console.log('then2: ' + data);
  return 'then2';
});
