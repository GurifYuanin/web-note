深拷贝


function cloneDeep(obj, hash = new WeakMap()) {
  if (Object.prototype.toString.call(obj).toLowerCase() === '[object object]') {
    // hash 用于避免循环调用
    if (hash.has(obj)) {
      return hash.get(obj);
    }
    const cloneObj = new obj.constructor();
    hash.set(obj, cloneObj);
    for (const key in obj) {
      if (obj.hasOwnProperty(key)) {
        cloneObj[key] = cloneDeep(obj[key], hash);
      }
    }
    return cloneObj;
  } else if (Array.isArray(obj)) {
    return obj.map(el => cloneDeep(el));
  } else {
    return obj;
  }
}

查找数组中重复的元素

先排序再遍历


function findDuplicate(arr) {
  const result = [];
  const sortedArr = arr.map(i => i).sort();
  for (let i = 0; i < sortedArr.length - 1; i++) {
    if (sortedArr[i] === sortedArr[i + 1] && !result.includes(sortedArr[i])) {
      result.push(sortedArr[i]);
    }
  }
  return result;
}

暴力解法


function findDuplicate(arr) {
  const result = [];
  const cloneArr = arr.map(i => i);
  while(cloneArr.length) {
    const el = cloneArr.pop();
    if (cloneArr.includes(el)) {
      result.push(el);
      cloneArr = cloneArr.filter(i => el !== i);
    }
  }
  return result;
}

升序数组中找到元素的重复次数 / 二分查找


function getCount(arr, el) {
  let result = 0;
  let idx = -1;
  for (let i = 0, j = arr.length - 1; i <= j;) {
    idx = Math.floor((i + j) / 2);
    if (el > arr[i]) {
      i = idx + 1;
    } else if (el < arr[j]) {
      j = idx - 1;
    } else {
      break;
    }
  }
  if (idx === -1) {
    return result;
  }
  let tmp = idx;
  while (arr[tmp] === el && tmp >= 0) {
    tmp--;
    result++;
  }
  tmp = idx + 1;
  while (arr[tmp] === el && tmp < arr.length) {
    tmp++;
    result++;
  }
  return result;
}
const n = getCount([ 1, 2, 3, 4, 5, 5, 5, 5, 6, 7, 10, 10 ], 10);

扁平化数组


function flattenDeep(arr, result = []) {
  if (Array.isArray(arr)) {
    for (const el of arr) {
      flattenDeep(el, result);
    }
  } else {
    result.push(arr);
  }
  return result;
}

函数柯里化

实现 sum(1, 2, 3) 等于 sum(1)(2)(3)，涉及知识点：

  • 函数柯里化
  • 闭包

const sum = (() => {
  let list = [];
  const add = function(...args) {
    list = list.concat(args);
    return add;
  }
  add.toString = function() {
    let sum = list.reduce((a, b) => a + b, 0);
    list = [];
    return sum;
  }
  return add;
})();

console.log(sum(1, 2 ,3).toString());
console.log(sum(1)(2)(3).toString());
