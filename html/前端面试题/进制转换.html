进制转换

问题描述

实现从 n 进制转化为 m 进制。 


const chars = '0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ';

n 进制转化为 10 进制

功能等价于 Number.parseInt(str, radix) 


function parseInt(str, radix) {
  let result = 0;
  let base = 1; // radix^0
  for (const s of str.split('').reverse()) {
    const n = chars.indexOf(s);
    if (n === -1) {
      throw new Error();
    } else {
      result += n * base;
      base *= radix;
    }
  }
  return result;
}

10 进制转化为 m 进制


function toString(num, radix) {
  if (num === 0) {
    return '0';
  }
  let result = '';
  while (num > 0) {
    const n = num % radix;
    if (n >= chars.length) {
      throw new Error();
    }
    result = chars[n] + result;
    num = Math.floor(num / radix);
  }
  return result;
}
